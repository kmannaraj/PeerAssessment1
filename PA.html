# Are there differences in activity patterns between weekdays and weekends?


day <- function(dates) {
  f <- function(date) {
    if (weekdays(date) %in% c("Saturday", "Sunday")) {
      "weekend"
    }
    else {
      "weekday"
    }
  }
  sapply(dates, f)
}

activity$day <- as.factor(day(activity$date))
